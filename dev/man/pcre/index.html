Parsing Regular Expressions and constructing an equivalent CombinedParser

This guide demonstrates constructing a recursive parser with the push! to Either technique by means of a similified parser for regular expressions.

This page outlines parts of the CombinedParser used in the re_str macro. CombinedParsers.jl is tested and benchmarked against the PCRE C library testset, see compliance report.

Characters and Escaped Characters

The simplest regular expression pattern is a character matcher.

using CombinedParsers
using CombinedParsers.Regexp
import CombinedParsers.Regexp: meta_chars

char_matcher =  Either(

    with_name("character",
        CharNotIn([ c for c in meta_chars])),

    with_name("escaped meta",
        Sequence(2, # emit unescaped
            '\\', CharIn(meta_chars)))

)
|🗄... Either
├─ [^\\\^\$\.\[\|\(\)\?\*\+\{] CharNotIn |> with_name(:character)
└─ 🗄 Sequence |> map(#49) |> with_name(:escaped meta)
   ├─ \\ 
   └─ [\\\^\$\.\[\|\(\)\?\*\+\{] CharIn
::Char

The char_matcher unescapes regex meta_chars:

julia> parse(char_matcher,"a"; log=true)
   match character: a
                    ^
'a': ASCII/Unicode U+0061 (category Ll: Letter, lowercase)

julia> parse(char_matcher,"\\["; log=true)
   match escaped meta: \\[
                       ^_^
'[': ASCII/Unicode U+005b (category Ps: Punctuation, open)

Transforming a Parsing

The parser for escaped and not escaped regular expression characters char_matcher emits Chars

julia> result_type(char_matcher)
Char

The Julia map method for CombinedParser creates a Transformation parser. The char_matcher is transformed to emit an CharIn matching the unescaped character:

julia> char_matcher_parser = map(CharIn,char_matcher);

julia> result_type(char_matcher_parser)
CharIn

A parser for a single character is constructed that can then be used to parse that specific Char

julia> p = parse(char_matcher_parser,"\\[")
re"\["

julia> parse(p,"[")
'[': ASCII/Unicode U+005b (category Ps: Punctuation, open)

Repeated patterns

Number of Repetitions

A basic feature of regular expression is whether a repeatable pattern (e.g. a char) is optional ?, or should be repeated at least once +, any times *, or {min,max} times.

The Pair syntax abbreviates mapping a match to a constant value. The do syntax is supported for all CombinedParsers constructors to implicitly call map.

import CombinedParsers.Regexp: integer

# https://www.pcre.org/original/doc/html/pcrepattern.html#SEC17
@with_names repetition = Either(
    "?" => (0,1), # == map(v->(0,1), parser("?"))
    "+" => (1,typemax(Int)),
    "*" => (0,typemax(Int)),
    Sequence(
        "{",
        integer,
        Optional( Sequence(2,",",
            Optional(integer, default=typemax(Int)))),
        "}") do v
            if v[3] isa Missing
                (v[2],v[2])
            else
                (v[2],v[3])
            end::Tuple{Int,Int}
    end
)
|🗄... Either |> with_name(:repetition)
├─ \?  |> map(Constant((0, 1)))
├─ \+  |> map(Constant((1, 9223372036854775807)))
├─ \*  |> map(Constant((0, 9223372036854775807)))
└─ 🗄 Sequence |> map(#41)
   ├─ \{ 
   ├─ 🗄 Sequence |> map(#41)
   │  ├─ \-?  |> Optional(default=missing)
   │  └─ [0-9]+ CharIn |> Repeat |> map(#41)
   ├─ 🗄? Sequence |> map(#49) |> Optional(default=missing)
   │  ├─ , 
   │  └─ 🗄? Sequence |> map(#41) |> Optional(default=9223372036854775807)
   │     ├─ \-?  |> Optional(default=missing)
   │     └─ [0-9]+ CharIn |> Repeat |> map(#41)
   └─ \} 
::Tuple{Int64,Int64}

repetition parses PCRE repetition syntax into a (min, max):

julia> parse(repetition, "?")
(0, 1)

julia> parse(repetition, "*")
(0, 9223372036854775807)

julia> parse(repetition, "+")
(1, 9223372036854775807)

julia> parse(repetition, "{2}")
(2, 2)

julia> parse(repetition, "{5,}")
(5, 9223372036854775807)

Either

Regular expressions can repeat character matchers and other sub-patterns when appending the repetition suffix.

repeatable = Either{CombinedParser}(char_matcher_parser)

We will add capture groups and other sub-patterns to repeatable later.

Repeatable patterns, Optional, Lazy and Atomic

The CombinedParser to parse a repeated pattern from PCRE syntax

@with_names quantified=Sequence(
        repeatable,
        Optional(repetition, default=(1,1)),
        Optional(map(v->convert(Char,v),with_name("possessive_quantifier",CharIn('+','?'))))
    ) do v
        pat = v[1]
        result = if v[2]==(1,1)
            pat
        elseif v[2]==(0,1)
            Optional(pat)
        else
            Repeat(pat,v[2])
        end
        if v[3] === missing
            result
        elseif v[3]=='+'
            Atomic(result)
        elseif v[3]=='?'
            Lazy(result)
        else
            result
        end
    end

quantified parses PCRE repetition syntax:

julia> p = parse(quantified, raw"\++?", log=true)
   match escaped meta: \\++?
                       ^_^
   match repetition: \\++?
                        ^
   match possessive_quantifier: \\++?
                                    ^
   match quantified: \\++?
                     ^___^
\+{1,9223372036854775807}?  |> Repeat |> Lazy
::Array{Char,1}

julia> parse(p, "+++")
1-element Array{Char,1}:
 '+'

Sequences and Alternations

Regular expression patterns can be written in sequence, delimited by | for matching either one of several alternatives.

@with_names begin
    sequence = map( v->sSequence(v...),
        Repeat( quantified ))

    alternation = Sequence(
        sequence,
        Repeat(Sequence(2, '|',sequence))) do v
            sEither(v[1],v[2]...)::CombinedParser
        end
end

The alternation parser transforms a regex string into a parser

julia> result_type(alternation)
CombinedParser

julia> p = parse(alternation, "ab|c", log=(:sequence, :alternation))
   match sequence: ab|c
                   ^^
   match sequence: ab|c
                      ^
   match alternation: ab|c
                      ^__^
|🗄... Either
├─ 🗄 Sequence
│  ├─ a
│  └─ b
└─ c
::Union{Char, Tuple{Char,Char}}

julia> parse(p, "c")
'c': ASCII/Unicode U+0063 (category Ll: Letter, lowercase)

Captures

Parentheses allow groupings that are repeatable.

import CombinedParsers.Regexp: name

@with_names begin
    noncapture_group=Sequence( 2, "(?:",alternation,")")

    capture_group=Sequence("(",alternation,")") do v
                Capture(v[2])
    end;
end

The push! to Either technique allows for the construction of recursive parsers.

push!(repeatable,capture_group);
push!(repeatable,noncapture_group);
julia> p = parse(alternation, "(ab|c)+", log=(:capture_group, repeatable))
   match capture_group: (ab|c)+
                        ^____^
(|🗄...){1,9223372036854775807} Either |> Capture -1 |> Repeat
├─ 🗄 Sequence
│  ├─ a
│  └─ b
└─ c
::Array{Union{Char, Tuple{Char,Char}},1}

julia> parse(p, "cabcabab")
5-element Array{Union{Char, Tuple{Char,Char}},1}:
 'c'
 ('a', 'b')
 'c'
 ('a', 'b')
 ('a', 'b')

Lookahead and Lookbehind

@with_names lookahead=Sequence(
    2,
    "(",
    Either(Sequence(
            v -> LookAhead(true,v[2]),
            Either("?=","*positive_lookahead:","*pla:"),
            alternation),
        Sequence(
            v -> LookAhead(false,v[2]),
            Either("?!","*negative_lookahead:","*nla:"),
            alternation)),
    ")");
@with_names lookbehind=Sequence(
    2,
    "(",
    Either(Sequence(v -> LookBehind(true,v[2]),
            Either("?<=","*positive_lookbehind:","*plb:"),alternation),
        Sequence(v -> LookBehind(false,v[2]),
            Either("?<!","*negative_lookbehind:","*nlb:"),alternation)),
    ")");

pushfirst!(repeatable,lookahead);
pushfirst!(repeatable,lookbehind);

nothing # hide
